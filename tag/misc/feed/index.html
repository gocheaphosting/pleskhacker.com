Column Descriptions
param – parameter keyword
val – value that corresponds to the parameter

Table Structure

CREATE TABLE `misc` (
 `param` varchar(255) character set ascii collate ascii_bin NOT NULL default '',
 `val` varbinary(255) NOT NULL default '                                                                                                                                                                                                                                                               ',
 PRIMARY KEY  (`param`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1
