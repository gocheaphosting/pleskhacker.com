SELECT domains.name, domains.real_size, Limits.value
FROM domains, Limits
WHERE domains.limits_id = Limits.id
AND domains.real_size > Limits.value
AND limit_name = 'disk_space'
AND Limits.value != -1
ORDER BY domains.name ASC;

ORDER BY sys_users.home ASC;
